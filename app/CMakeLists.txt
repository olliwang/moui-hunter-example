cmake_minimum_required(VERSION 3.4.1)

add_library(app
    STATIC
    "application.cc"
    "demo_view.cc"
    "main.cc")

set_target_properties(app
    PROPERTIES
    CXX_STANDARD 11
    CXX_STANDARD_REQUIRED YES
    CXX_EXTENSIONS NO)

target_include_directories(app
    PRIVATE
    ".."
    "../deps")

if(ANDROID)
    add_subdirectory("../deps/moui/android" "moui-jni")
    target_link_libraries(app PUBLIC moui-jni)
else()
    add_subdirectory("../deps/moui/moui" "moui")
    target_link_libraries(app PUBLIC moui)
endif()

# Adds depdendency through hunter.
hunter_add_package(yaml-cpp)
find_package(yaml-cpp CONFIG REQUIRED)
target_link_libraries(app PUBLIC yaml-cpp::yaml-cpp)
